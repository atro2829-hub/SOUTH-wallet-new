   480→        });

   485→          useAppStore.getState().setUser({
   486→            ...storeUser,
   487→            [balanceKey]: txResult.snapshot.val(),
   488→          });
   489→        }
   490→      }

   493→      setCardCode('');

   523→        id: requestId,
   524→        userId: user.id,
   525→        userName: user.name,
   526→        amount: depositAmountNum + promoDiscount,
   527→        currency: depositCurrency,
   528→        method: depositMethod,

   531→        notes: promoApplied ? `كود خصم مطبق: ${promoCode}` : '',
   532→        createdAt: new Date().toISOString(),
   533→      };
   534→
   535→      // Add crypto-specific fields
   536→      if (depositMethod === 'crypto') {
   537→        const selectedCrypto = cryptoCurrencies.find(c => c.id === selectedCryptoId);
   538→        const activeNetworkName = selectedCryptoNetwork || selectedCrypto?.network || '';
   539→        const activeNetworkData = selectedCrypto?.networks?.find(n => n.networkName === activeNetworkName);
   540→        const depositAddress = activeNetworkData?.walletAddress || selectedCrypto?.address || '';
   541→        request.cryptoId = selectedCryptoId;
   542→        request.cryptoSymbol = selectedCrypto?.symbol || '';
   543→        request.cryptoNetwork = activeNetworkName;
