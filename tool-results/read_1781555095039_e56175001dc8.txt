    92→// Helper: mask account number (show first 3 and last 4 digits)
    93→function maskAccountNumber(accNum: string): string {
    94→  if (!accNum || accNum.length <= 7) return accNum;
    95→  return accNum.substring(0, 3) + '****' + accNum.substring(accNum.length - 4);
