    10→    10→import { Button } from '@/components/ui/button';

    17→    17→

    22→    22→  const [loading, setLoading] = useState(true);
    23→    23→  const [saving, setSaving] = useState(false);
    24→    24→  const [search, setSearch] = useState('');
    25→    25→  const [dialog, setDialog] = useState(false);
    26→    26→  const [packageDialog, setPackageDialog] = useState(false);

    29→    29→  const [parentService, setParentService] = useState<string | null>(null);
    30→    30→  const [expandedService, setExpandedService] = useState<string | null>(null);
    31→    31→
    32→    32→  // Form state
    33→    33→  const [name, setName] = useState('');
    34→    34→  const [description, setDescription] = useState('');
    35→    35→  const [icon, setIcon] = useState('');
    36→    36→  const [color, setColor] = useState('#5C1A1B');
    37→    37→  const [categoryId, setCategoryId] = useState('wallet-services');
    38→    38→  const [sectionId, setSectionId] = useState('wallet-services');
    39→    39→  const [inputLabel, setInputLabel] = useState('معرف العميل');
    40→    40→  const [inputType, setInputType] = useState('text');
    41→    41→  const [inputPrefix, setInputPrefix] = useState('');
    42→    42→  const [isActive, setIsActive] = useState(true);
    43→    43→  const [sortOrder, setSortOrder] = useState(0);
    44→    44→
    45→    45→  // Package form state
    46→    46→  const [pkgName, setPkgName] = useState('');
    47→    47→  const [pkgPrice, setPkgPrice] = useState(0);
    48→    48→  const [pkgCurrency, setPkgCurrency] = useState<'YER' | 'SAR' | 'USD'>('YER');
    49→    49→  const [pkgCostPrice, setPkgCostPrice] = useState(0);
    50→    50→  const [pkgCommission, setPkgCommission] = useState(0);
    51→    51→  const [pkgCommissionType, setPkgCommissionType] = useState<'percentage' | 'fixed'>('percentage');
    52→    52→  const [pkgExecutionType, setPkgExecutionType] = useState<'manual' | 'auto'>('manual');
    53→    53→  const [pkgApiProviderId, setPkgApiProviderId] = useState('');
    54→    54→  const [pkgApiProductId, setPkgApiProductId] = useState('');
    55→    55→  const [pkgIsActive, setPkgIsActive] = useState(true);
    56→    56→  const [pkgSortOrder, setPkgSortOrder] = useState(0);
    57→    57→  const [pkgDescription, setPkgDescription] = useState('');
    58→    58→
    59→    59→  // Listen to wallet services

    72→    72→
    73→    73→  // Listen to sections
    74→    74→  useEffect(() => {

    78→    78→    });

    86→    86→  };

    90→    90→    setPkgCommissionType('percentage'); setPkgExecutionType('manual'); setPkgApiProviderId('');
    91→    91→    setPkgApiProductId(''); setPkgIsActive(true); setPkgSortOrder(0); setPkgDescription('');
    92→    92→    setEditingPkg(null); setParentService(null);
    93→    93→  };
    94→    94→
    95→    95→  const handleIconUpload = async (e: React.ChangeEvent<HTMLInputElement>) => {
    96→    96→    const file = e.target.files?.[0];
    97→    97→    if (!file) return;
    98→    98→    if (file.size > 2000000) {
    99→    99→      showToast('حجم الصورة يجب أن يكون أقل من 2MB', 'error');
   100→   100→      return;
   101→   101→    }
   102→   102→    const reader = new FileReader();
   103→   103→    reader.onload = async (ev) => {
   104→   104→      try {
   105→   105→        const base64 = ev.target?.result as string;
   106→   106→        // Compress to max 128x128 for Firebase RTDB size limits and user app performance
   107→   107→        const compressed = await compressBase64Image(base64, 128, 0.7);
   108→   108→        setIcon(compressed);
   109→   109→        showToast('تم رفع الأيقونة بنجاح', 'success');
   110→   110→      } catch (err) {
   111→   111→        console.error('[WalletServicesPanel] Icon compression failed:', err);
   112→   112→        // Fallback to original if compression fails
   113→   113→        setIcon(ev.target?.result as string);
   114→   114→        showToast('تم رفع الأيقونة (بدون ضغط)', 'info');
   115→   115→      }
   116→   116→    };
   117→   117→    reader.readAsDataURL(file);
   118→   118→  };
   119→   119→
   120→   120→  const handleSave = async () => {
   121→   121→    if (!name.trim()) { showToast('يرجى إدخال الاسم', 'error'); return; }
   122→   122→    setSaving(true);
   123→   123→    try {
   124→   124→      const serviceId = editing || name.trim().replace(/[\s]+/g, '-').replace(/[^\u0600-\u06FFa-zA-Z0-9\-]/g, '').toLowerCase() || `service-${Date.now()}`;
   125→   125→
   126→   126→      const data = {
   127→   127→        id: serviceId, name: name.trim(), description: description.trim(), icon, color, categoryId,
   128→   128→        sectionId, inputLabel: inputLabel.trim(), inputType, inputPrefix: inputPrefix.trim(),
   129→   129→        isActive, sortOrder, updatedAt: new Date().toISOString(),
   130→   130→        createdAt: editing && services[editing]?.createdAt ? services[editing].createdAt : new Date().toISOString(),
   131→   131→      };
   132→   132→
   133→   133→      // Use update() instead of set() to avoid overwriting nested data like packages
   134→   134→      await update(ref(database, `walletServices/${serviceId}`), data);
   135→   135→
   136→   136→      // Also sync to providers path for compatibility with user app
   137→   137→      // Use update() to avoid overwriting existing provider fields that may have been set elsewhere
   138→   138→      await update(ref(database, `providers/${serviceId}`), {
   139→   139→        id: serviceId, name: name.trim(), color, icon, categoryId, sectionId,
   140→   140→        inputLabel: inputLabel.trim(), inputType, inputPrefix: inputPrefix.trim(), isActive,
   141→   141→        updatedAt: new Date().toISOString(),
   142→   142→      });
   143→   143→
   144→   144→      // Update visibility
   145→   145→      await update(ref(database, `adminSettings/visibility/providers/${serviceId}`), {});
   146→   146→
   147→   147→      showToast(editing ? 'تم تحديث الخدمة بنجاح' : 'تم إضافة الخدمة بنجاح', 'success');
   148→   148→      setDialog(false);
   149→   149→      resetForm();
   150→   150→    } catch (e) {
   151→   151→      console.error('[WalletServicesPanel] handleSave error:', e);
   152→   152→      showToast(`حدث خطأ أثناء الحفظ: ${e instanceof Error ? e.message : 'خطأ غير معروف'}`, 'error');
   153→   153→    } finally {
   154→   154→      setSaving(false);
   155→   155→    }
   156→   156→  };
   157→   157→
   158→   158→  const handleSavePkg = async () => {

   160→   160→    setSaving(true);
   161→   161→    try {
   162→   162→      const pkgId = editingPkg || `pkg-${Date.now()}-${Math.random().toString(36).substring(2, 6)}`;
   163→   163→      const data = {
   164→   164→        id: pkgId, name: pkgName.trim(), price: pkgPrice, currency: pkgCurrency,
   165→   165→        costPrice: pkgCostPrice, commission: pkgCommission, commissionType: pkgCommissionType,
   166→   166→        executionType: pkgExecutionType, apiProviderId: pkgApiProviderId.trim(),
   167→   167→        apiProductId: pkgApiProductId.trim(), isActive: pkgIsActive, sortOrder: pkgSortOrder,
   168→   168→        description: pkgDescription.trim(),
   169→   169→      };
   170→   170→
   171→   171→      await set(ref(database, `walletServices/${parentService}/packages/${pkgId}`), data);
   172→   172→      showToast(editingPkg ? 'تم تحديث الباقة' : 'تم إضافة الباقة', 'success');
   173→   173→      setPackageDialog(false);
   174→   174→      resetPkgForm();
   175→   175→    } catch (e) {
   176→   176→      console.error('[WalletServicesPanel] handleSavePkg error:', e);
   177→   177→      showToast('حدث خطأ أثناء الحفظ', 'error');
   178→   178→    } finally {
   179→   179→      setSaving(false);
   180→   180→    }
   181→   181→  };
   182→   182→
   183→   183→  const handleDelete = async (id: string) => {
   184→   184→    if (!confirm('هل أنت متأكد من حذف هذه الخدمة؟')) return;
   185→   185→    try {
   186→   186→      await remove(ref(database, `walletServices/${id}`));
   187→   187→      await remove(ref(database, `providers/${id}`));
   188→   188→      await remove(ref(database, `adminSettings/visibility/providers/${id}`));
   189→   189→      showToast('تم حذف الخدمة بنجاح', 'success');
   190→   190→    } catch (e) {
   191→   191→      console.error('[WalletServicesPanel] handleDelete error:', e);
   192→   192→      showToast('حدث خطأ أثناء الحذف', 'error');
   193→   193→    }
   194→   194→  };
   195→   195→
   196→   196→  const handleDeletePkg = async (serviceId: string, pkgId: string) => {
   197→   197→    try {
   198→   198→      await remove(ref(database, `walletServices/${serviceId}/packages/${pkgId}`));
   199→   199→      showToast('تم حذف الباقة', 'success');
   200→   200→    } catch (e) {
   201→   201→      console.error('[WalletServicesPanel] handleDeletePkg error:', e);
   202→   202→      showToast('حدث خطأ أثناء الحذف', 'error');
   203→   203→    }
   204→   204→  };
   205→   205→
   206→   206→  const handleToggleActive = async (serviceId: string, active: boolean) => {
   207→   207→    try {
   208→   208→      const now = new Date().toISOString();
   209→   209→      // Update walletServices path
   210→   210→      await update(ref(database, `walletServices/${serviceId}`), { isActive: active, updatedAt: now });
   211→   211→      // Also sync isActive to providers path so user app sees the change
   212→   212→      await update(ref(database, `providers/${serviceId}`), { isActive: active, updatedAt: now });
   213→   213→      showToast(active ? 'تم تفعيل الخدمة' : 'تم تعطيل الخدمة', 'success');
   214→   214→    } catch (e) {
   215→   215→      console.error('[WalletServicesPanel] handleToggleActive error:', e);
   216→   216→      showToast('حدث خطأ أثناء تحديث الحالة', 'error');
   217→   217→    }
   218→   218→  };
   219→   219→
   220→   220→  const sectionOptions = [
   221→   221→    { value: 'wallet-services', label: 'خدمات المحفظة' },
   222→   222→    { value: 'entertainment', label: 'الخدمات الترفيهية' },
   223→   223→    { value: 'providers', label: 'مزودين الخدمات' },

   228→   228→    { value: 'crypto', label: 'الكريبتو' },
   229→   229→    ...Object.entries(sections).map(([id, s]: [string, any]) => ({ value: id, label: s.name || id })),
   230→   230→  ];
   231→   231→
   232→   232→  const serviceList = Object.entries(services).map(([id, s]) => ({ id, ...s }));
   233→   233→  const filteredServices = serviceList.filter(s => !search || s.name?.includes(search) || s.description?.includes(search));
   234→   234→
   235→   235→  if (loading) return <div className="flex items-center justify-center min-h-[400px]"><div className="w-8 h-8 border-2 border-purple-500 border-t-transparent rounded-full animate-spin" /></div>;
   236→   236→
   237→   237→  return (
   238→   238→    <div className="space-y-6">
   239→   239→      <div className="flex items-center justify-between">
   240→   240→        <div>
   241→   241→          <h1 className="text-2xl font-bold">خدمات المحفظة</h1>
   242→   242→          <p className="text-muted-foreground text-sm mt-1">{formatNumber(serviceList.length)} خدمة مخصصة</p>

   248→   248→
   249→   249→      <div className="relative">
   250→   250→        <Search className="absolute right-3 top-1/2 -translate-y-1/2 w-4 h-4 text-muted-foreground" />

   252→   252→      </div>
   253→   253→
   254→   254→      <div className="space-y-3 max-h-[calc(100vh-280px)] overflow-y-auto scrollbar-thin">
   255→   255→        {filteredServices.map((svc, i) => {
   256→   256→          const isExpanded = expandedService === svc.id;
   257→   257→          const packages = svc.packages || {};
   258→   258→          const pkgCount = Object.keys(packages).length;
   259→   259→          const sectionLabel = sectionOptions.find(s => s.value === svc.sectionId)?.label || svc.sectionId;
   260→   260→
   261→   261→          return (
   262→   262→            <motion.div key={svc.id} initial={{ opacity: 0 }} animate={{ opacity: 1 }} transition={{ delay: i * 0.02 }}>
   263→   263→              <Card className="admin-card border-0 shadow-none">
   264→   264→                <CardContent className="p-4">
   265→   265→                  <div className="flex items-center justify-between">
   266→   266→                    <div className="flex items-center gap-3 min-w-0 flex-1">
   267→   267→                      {svc.icon ? (
   268→   268→                        <img src={svc.icon} className="w-10 h-10 rounded-lg object-cover shrink-0" alt="" />
   269→   269→                      ) : (
   270→   270→                        <div className="w-10 h-10 rounded-lg flex items-center justify-center shrink-0" style={{ backgroundColor: (svc.color || '#5C1A1B') + '20' }}>
   271→   271→                          <Wallet className="w-5 h-5" style={{ color: svc.color || '#5C1A1B' }} />

   282→   282→                      </div>

   284→   284→                    <div className="flex items-center gap-1.5 shrink-0">
   285→   285→                      <Switch checked={svc.isActive !== false} onCheckedChange={(v) => handleToggleActive(svc.id, v)} />

   287→   287→                        setParentService(svc.id);
   288→   288→                        resetPkgForm();
   289→   289→                        setPackageDialog(true);
   290→   290→                      }} title="إضافة باقة">
   291→   291→                        <PackagePlus className="w-4 h-4" />
   292→   292→                      </Button>
   293→   293→                      <Button variant="ghost" size="sm" onClick={() => setExpandedService(isExpanded ? null : svc.id)}>

   295→   295→                      </Button>
   296→   296→                      <Button variant="ghost" size="sm" onClick={() => {
   297→   297→                        setEditing(svc.id);
   298→   298→                        setName(svc.name || ''); setDescription(svc.description || ''); setIcon(svc.icon || '');
   299→   299→                        setColor(svc.color || '#5C1A1B'); setCategoryId(svc.categoryId || 'wallet-services');
   300→   300→                        setSectionId(svc.sectionId || 'wallet-services');
   301→   301→                        setInputLabel(svc.inputLabel || 'معرف العميل'); setInputType(svc.inputType || 'text');
   302→   302→                        setInputPrefix(svc.inputPrefix || ''); setIsActive(svc.isActive !== false);
   303→   303→                        setSortOrder(svc.sortOrder || 0); setDialog(true);
   304→   304→                      }}><Edit className="w-4 h-4" /></Button>
   305→   305→                      <Button variant="ghost" size="sm" onClick={() => handleDelete(svc.id)}><Trash2 className="w-4 h-4 text-red-500" /></Button>

   307→   307→                  </div>
   308→   308→
   309→   309→                  {/* Expanded: Show packages */}
   310→   310→                  <AnimatePresence>
   311→   311→                    {isExpanded && (
   312→   312→                      <motion.div initial={{ height: 0, opacity: 0 }} animate={{ height: 'auto', opacity: 1 }} exit={{ height: 0, opacity: 0 }} className="overflow-hidden">
   313→   313→                        <div className="mt-3 pt-3 border-t border-border/50 space-y-2 max-h-60 overflow-y-auto scrollbar-thin">
   314→   314→                          {pkgCount > 0 ? (
   315→   315→                            Object.entries(packages).map(([pkgId, pkg]: [string, any]) => (
   316→   316→                              <div key={pkgId} className="flex items-center justify-between p-2 rounded-lg bg-muted/30">
   317→   317→                                <div className="flex items-center gap-2 min-w-0 flex-1">
   318→   318→                                  <Package className="w-4 h-4 text-muted-foreground shrink-0" />
   319→   319→                                  <div className="min-w-0 flex-1">
   320→   320→                                    <p className="text-xs font-medium truncate">{pkg.name}</p>
   321→   321→                                    <div className="flex items-center gap-2 mt-0.5">

   323→   323→                                      {pkg.commission > 0 && <span className="text-[10px] text-green-600">عمولة: {pkg.commission}{pkg.commissionType === 'percentage' ? '%' : ''}</span>}
   324→   324→                                      {pkg.executionType === 'auto' && <Badge className="bg-blue-500/20 text-blue-600 text-[9px] py-0">تلقائي</Badge>}
   325→   325→                                    </div>
   326→   326→                                  </div>
   327→   327→                                </div>
   328→   328→                                <div className="flex items-center gap-1 shrink-0">
   329→   329→                                  <Badge className={pkg.isActive !== false ? 'bg-green-500/20 text-green-600 text-[9px] py-0' : 'bg-red-500/20 text-red-500 text-[9px] py-0'}>
   330→   330→                                    {pkg.isActive !== false ? 'نشط' : 'معطل'}
   331→   331→                                  </Badge>
   332→   332→                                  <Button variant="ghost" size="sm" className="h-7 w-7 p-0" onClick={() => {
   333→   333→                                    setParentService(svc.id);
   334→   334→                                    setEditingPkg(pkgId);
   335→   335→                                    setPkgName(pkg.name || ''); setPkgPrice(pkg.price || 0);
   336→   336→                                    setPkgCurrency(pkg.currency || 'YER'); setPkgCostPrice(pkg.costPrice || 0);
   337→   337→                                    setPkgCommission(pkg.commission || 0); setPkgCommissionType(pkg.commissionType || 'percentage');
   338→   338→                                    setPkgExecutionType(pkg.executionType || 'manual');
   339→   339→                                    setPkgApiProviderId(pkg.apiProviderId || ''); setPkgApiProductId(pkg.apiProductId || '');
   340→   340→                                    setPkgIsActive(pkg.isActive !== false); setPkgSortOrder(pkg.sortOrder || 0);
   341→   341→                                    setPkgDescription(pkg.description || '');
